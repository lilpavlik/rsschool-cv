        class Worker extends Thread {
            double speed;
            int acceptedClients;
            int id;
            Clients clients;
        
            Worker(double speed, Clients clients, int id) {
                this.speed = speed;
                this.clients = clients;
                this.id = id;
            }
        
            public int getAcceptedClients() {
                return acceptedClients;
            }
        
            @Override
            public void run() {
                System.out.printf("Работник №%d приступил к работе!\nСкорость обслуживания: %.0f мс\n\n", id, speed);
                while (clients.getAmount() > 0) {
                    try {
                        Thread.sleep((long) speed);
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                    if (clients.getAmount() < 1) {
                        break;
                    }
                    clients.amount--;
                    acceptedClients++;
                    System.out.printf("[Работник №%d] > осталось обслужить %d посетителей\n", id, clients.amount);
                }
            }
